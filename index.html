🔧 One-Time Google Sheets Setup

Connect your Google Sheet so all devices share the same live price list. Takes about 5 minutes.

  1. Open your Enfield prices Google Sheet (or create one with columns: department, treatment, original_price, offer_price, 3_sessions, 6_sessions, notes)
  2. In Google Sheets click Extensions → Apps Script
  3. Delete any existing code and paste the script below, then click Save
  4. Click Deploy → New Deployment → Web App
  5. Set Execute as: Me · Set Who has access: Anyone
  6. Click Deploy → copy the Web App URL → paste it below
function doGet(e) {
  var sheetId = PropertiesService.getScriptProperties().getProperty("SHEET_ID"); 
var sheet = SpreadsheetApp.openById(sheetId).getActiveSheet();
  var data = sheet.getDataRange().getValues();
  var headers = data[0];
  var rows = data.slice(1).map(function(row) {
    var obj = {};
    headers.forEach(function(h, i) { obj[h] = row[i] ? String(row[i]) : null; });
    return obj;
  });
  return ContentService.createTextOutput(JSON.stringify({status:'ok',data:rows}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  var sheetId = PropertiesService.getScriptProperties().getProperty("SHEET_ID"); var sheet = SpreadsheetApp.openById(sheetId).getActiveSheet();
  var payload = JSON.parse(e.postData.contents);
  if (payload.action === 'update') {
    var data = sheet.getDataRange().getValues();
    var headers = data[0];
    var deptIdx = headers.indexOf('department');
    var nameIdx = headers.indexOf('treatment');
    var found = false;
    for (var i = 1; i < data.length; i++) {
      if (data[i][deptIdx] === payload.department && data[i][nameIdx] === payload.treatment) {
        headers.forEach(function(h, j) { if (payload[h] !== undefined) sheet.getRange(i+1, j+1).setValue(payload[h]||''); });
        found = true; break;
      }
    }
    if (!found) {
      var newRow = headers.map(function(h) { return payload[h] || ''; });
      sheet.appendRow(newRow);
    }
    return ContentService.createTextOutput(JSON.stringify({status:'ok'})).setMimeType(ContentService.MimeType.JSON);
  }
  if (payload.action === 'delete') {
    var data2 = sheet.getDataRange().getValues();
    var headers2 = data2[0];
    var dIdx = headers2.indexOf('department');
    var nIdx = headers2.indexOf('treatment');
    for (var j = 1; j < data2.length; j++) {
      if (data2[j][dIdx] === payload.department && data2[j][nIdx] === payload.treatment) {
        sheet.deleteRow(j+1); break;
      }
    }
    return ContentService.createTextOutput(JSON.stringify({status:'ok'})).setMimeType(ContentService.MimeType.JSON);
  }
}
